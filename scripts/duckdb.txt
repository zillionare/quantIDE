-- 创建qmt缓存下载任务错误表
CREATE TABLE "bars_cache_failure"  (
  "id" integer not null primary key,
  "symbol" integer not null,
  "frame_type" integer not null,
  "frame" date not null,
  "reason" integer not null
);


create index symbol_frame_idx on bars_cache_failure (symbol, frame_type, frame);

-- 创建qmt缓存下载任务表
create table if not exists bars_cache_status (
    "frame_type" string primary key, --记录属于哪个周期。1分钟和1天允许不一样的起点
    "start" date,
    "stop" date
);


-- 创建clickhouse行情同步表
CREATE SEQUENCE IF NOT EXISTS bars_sync_status_id;

create table if not exists bars_sync_status (
    "id" INTEGER default nextval('bars_sync_status_id') primary key,
    "start" date,
    "stop" date,
    "frame_type" str,  --记录属于哪个周期。1分钟和1天允许不一样的起点
    "kind" str, -- 记录类型， 可以是ohlc, st, factor, limit_price, suspend, turnover
);

--创建clickhouse 板块同步表
CREATE SEQUENCE IF NOT EXISTS sync_secotrs_jobs_id;
create table if not exists sync_secotrs_jobs(
    "id" INTEGER default nextval('bars_sync_status_id') primary key,
    "frame" date, -- 日期
    "count" int -- 当天板块数
);

-- 创建 clickhouse ashares list 同步表
create table if not exists sync_ashare_list_status(
    "frame" date primary key, -- 日期
    "count" int -- 当天个股数
);

-- 创建系统配置表
create table if not exists sys(
    "key" varchar(20) primary key,
    "value" varchar primary key
)
